Executes invocables until any of them returns a valid result<T>, then terminates, returning that result<T>.

Otherwise, returns result<T> with the last invalid status. 

  auto r = pass(1, 2, 3)
         | all(
            [](int a, int b, int c) -> result<int> { return std::max({a, b, c}); },
            [](int a, int b, int c) -> result<int> { return std::min({a, b, c}); },
            [](int a, int b, int c) -> result<int> { return "failure"_msg; }
          );

  std::cout << std::boolalpha << static_cast<bool>(r) << std::endl;  // false

◆ any()

template<typename... InvocableTs>
constexpr decltype(auto) zen::any ( InvocableTs &&...  t)
constexpr

Executes invocables until any of them returns an invalid result<T>, then terminates, returning a result<T> holding the first invalid status.

Otherwise, returns a result<T> containing values returned by all invocables. 

  auto r = pass(1, 2, 3)
         | any(
            [](int a, int b, int c) -> result<int> { return std::max({a, b, c}); },
            [](int a, int b, int c) -> result<int> { return std::min({a, b, c}); },
            [](int a, int b, int c) -> result<int> { return "failure"_msg; }
          );

  std::cout << std::boolalpha << static_cast<bool>(r) << std::endl;  // true

◆ operator|()

template<typename T , typename Fn >
decltype(auto) zen::operator| ( result< T > &&  r,
Fn &&  f 
)

Chains together invocables which return a result<T>

Short-circuits when an invalid result<T> is produce, and returns a result<T> with the invalid status causing failure.

  auto r = pass(1, 2, 3)
         | [](int a, int b, int c) -> result<int> { return std::max({a, b, c}); }, // <-- executed
         | [](int a, int b, int c) -> result<int> { return "failure"_msg; }        // <-- execution stops here
           [](int a, int b, int c) -> result<int> { return std::min({a, b, c}); }; // <-- never invoked

  std::cout << std::boolalpha << static_cast<bool>(r) << std::endl;  // false
